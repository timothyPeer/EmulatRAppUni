// ============================================================================

// Alpha Exception Vector Architecture - Complete Overview

// ============================================================================

//

// SINGLE SOURCE OF TRUTH for exception handling in Alpha AXP emulator.

//

 

// ============================================================================

// 1. DESIGN PRINCIPLES

// ============================================================================

//

// A. Separation of Concerns:

//    - Pipeline: Describes WHAT happened (ExceptionClass + context)

//    - Mapping: Resolves HOW to handle it (PalVectorId)

//    - Preparation: Synchronizes state (HWPCB + IPRs)

//    - Delivery: Enters PAL mode (PC update)

//

// B. Single Source of Truth:

//    - mapClassToPalVector() is ONLY place that maps class to vector

//    - preparePendingEventForDelivery() is ONLY place that updates state

//    - ExceptionFactory functions are ONLY way to create events

//

// C. No Premature Resolution:

//    - Pipeline NEVER computes palVectorId

//    - FaultDispatcher resolves vectors during preparation

//    - CALL_PAL vectors CALCULATED, not stored

//

// D. Explicit ITB/DTB Distinction:

//    - No generic "MEMORY_FAULT" class

//    - Separate ExceptionClass for each TB domain

//    - Property_info.isInstruction distinguishes domains

 

// ============================================================================

// 2. DATA FLOW

// ============================================================================

//

//   ┌─────────────────────────────────────────────────────────────────┐

//   │ PIPELINE (IBox/MBox/FBox/EBox)                                  │

//   │                                                                  │

//   │  Detects fault → Use ExceptionFactory → Create PendingEvent     │

//   │                                                                  │

//   │  Example: DTB miss at 0x20000                                  │

//   │    auto ev = ExceptionFactory::makeDTBMissSingleEvent(          │

//   │        cpuId, 0x20000, isWrite);                                │

//   │                                                                  │

//   │  Event fields set:                                              │

//   │    - kind = Exception                                           │

//   │    - exceptionClass = DTB_MISS_SINGLE                           │

//   │    - faultVA = 0x20000                                          │

//   │    - palVectorId = INVALID (not resolved yet)                   │

//   └─────────────────────────────────────────────────────────────────┘

//                              ↓

//   ┌─────────────────────────────────────────────────────────────────┐

//   │ FAULT DISPATCHER                                                 │

//   │                                                                  │

//   │  Queue event → Arbitrate priority → Select winner               │

//   │                                                                  │

//   │  No vector computation here!                                     │

//   │  Just priority-based selection.                                  │

//   └─────────────────────────────────────────────────────────────────┘

//                              ↓

//   ┌─────────────────────────────────────────────────────────────────┐

//   │ PREPARATION (preparePendingEventForDelivery)                     │

//   │                                                                  │

//   │  1. Save PC/PS to HWPCB                                          │

//   │     hwpcb.setSavedPC(faultPC);                                   │

//   │     hwpcb.setSavedPS(faultPS);                                   │

//   │                                                                  │

//   │  2. Resolve vector via mapClassToPalVector()                     │

//   │     ev.palVectorId = mapClassToPalVector(ev);                    │

//   │     // DTB_MISS_SINGLE → PalVectorId::DTB_MISS_SINGLE (0x0300)  │

//   │                                                                  │

//   │  3. Calculate PAL entry PC                                       │

//   │     palEntryPC = pal_base + 0x0300;                              │

//   │                                                                  │

//   │  4. Update EXC_ADDR                                              │

//   │     iprs.exc_addr = palEntryPC;                                  │

//   │     hwpcb.setHWPCBExc_Addr(palEntryPC);                          │

//   │                                                                  │

//   │  5. Update exception-specific IPRs                               │

//   │     updateExceptionSummary(iprs, ev);     // EXC_SUM            │

//   │     updateMemoryManagementStatus(iprs, ev); // MM_STAT           │

//   │     saveFaultVirtualAddress(iprs, hwpcb, ev.faultVA); // VA     │

//   └─────────────────────────────────────────────────────────────────┘

//                              ↓

//   ┌─────────────────────────────────────────────────────────────────┐

//   │ DELIVERY (deliverException)                                      │

//   │                                                                  │

//   │  1. Update PC to PAL entry                                       │

//   │     hwpcb.setPC(iprs.exc_addr);  // Sets PALmode bit 0          │

//   │                                                                  │

//   │  2. Enter kernel mode                                            │

//   │     hwpcb.setCM(0);  // Kernel mode                              │

//   │                                                                  │

//   │  CPU now executing PAL handler at PAL_BASE + 0x0300             │

//   └─────────────────────────────────────────────────────────────────┘

 

// ============================================================================

// 3. CALL_PAL SPECIAL CASE

// ============================================================================

//

// CALL_PAL instructions encode function code in bits [7:0].

// PC calculation formula (Alpha AXP Architecture Manual 4.7.3):

//

//   PC[63:15] = PAL_BASE[63:15]

//   PC[14]    = 0

//   PC[13]    = 1

//   PC[12]    = function[7]

//   PC[11:6]  = function[5:0]

//   PC[5:1]   = 0

//   PC[0]     = 1 (PALmode)

//

// Validation rules:

//   - Invalid ranges: 0x40-0x7F, 0xC0-0xFF → OPCDEC

//   - Privileged (0x00-0x3F) requires kernel mode → OPCDEC if not kernel

//   - Unprivileged (0x80-0xBF) allowed from any mode

//

// Flow:

//   1. IBox decodes CALL_PAL instruction

//   2. Extract function code from bits [7:0]

//   3. Create event:

//        auto ev = ExceptionFactory::makeCallPalEvent(cpuId, pc, function);

//   4. Preparation validates and calculates PC:

//        if (!isValidCallPalFunction(function, mode)) {

//            // Convert to OPCDEC

//            ev.exceptionClass = ILLEGAL_OPCODE;

//            palEntryPC = pal_base + 0x0400;

//        } else {

//            // Calculate CALL_PAL entry

//            palEntryPC = calculateCallPalEntryPC(pal_base, function);

//        }

 

// ============================================================================

// 4. EXCEPTION CLASS HIERARCHY

// ============================================================================

//

// Memory Management (8 classes):

//   ITB_MISS            → 0x0580  (Instruction TB miss)

//   ITB_ACV             → 0x0480  (Instruction access violation)

//   DTB_MISS_SINGLE     → 0x0300  (Data TB miss, 1-level PT)

//   DTB_MISS_DOUBLE_3   → 0x0100  (Data TB miss, 3-level PT)

//   DTB_MISS_DOUBLE_4   → 0x0180  (Data TB miss, 4-level PT)

//   DTB_ACV             → 0x0380  (Data access violation)

//   DTB_FAULT           → 0x0380  (Data fault: FOE/FOR/FOW/sign check)

//

// Alignment & Opcodes (3 classes):

//   UNALIGNED           → 0x0280  (Unaligned access)

//   ILLEGAL_OPCODE      → 0x0400  (Invalid opcode)

//   FEN                 → 0x0200  (FP disabled)

//

// Arithmetic (2 classes):

//   ARITHMETIC          → 0x0600  (Integer/FP trap)

//   MT_FPCR             → 0x0700  (Move to FPCR)

//

// System (4 classes):

//   CALL_PAL            → calculated  (Generic PAL function)

//   BREAKPOINT          → calculated  (BPT, function 0x80)

//   BUGCHECK            → calculated  (BUGCHECK, function 0x81)

//   GENTRAP             → calculated  (GENTRAP, function 0xAA)

//

// Asynchronous (3 classes):

//   INTERRUPT           → 0x0680  (HW/SW/AST interrupt)

//   MACHINE_CHECK       → 0x0500  (Machine check)

//   RESET               → 0x0780  (Reset/wakeup)

 

// ============================================================================

// 5. VECTOR RESOLUTION MAP

// ============================================================================

//

// This is the SINGLE SOURCE OF TRUTH mapping:

//

// ExceptionClass          → PalVectorId           → Offset

// ─────────────────────────────────────────────────────────

// ITB_MISS                → ITB_MISS              → 0x0580

// ITB_ACV                 → IACV                  → 0x0480

// DTB_MISS_SINGLE         → DTB_MISS_SINGLE       → 0x0300

// DTB_MISS_DOUBLE_3       → DTB_MISS_DOUBLE_3     → 0x0100

// DTB_MISS_DOUBLE_4       → DTB_MISS_DOUBLE_4     → 0x0180

// DTB_ACV                 → DFAULT                → 0x0380

// DTB_FAULT               → DFAULT                → 0x0380

// UNALIGNED               → UNALIGN               → 0x0280

// ILLEGAL_OPCODE          → OPCDEC                → 0x0400

// FEN                     → FEN                   → 0x0200

// ARITHMETIC              → ARITH                 → 0x0600

// MT_FPCR                 → MT_FPCR               → 0x0700

// CALL_PAL                → INVALID (calculate)   → formula

// BREAKPOINT              → BREAKPOINT            → calculated

// BUGCHECK                → BUGCHECK              → calculated

// GENTRAP                 → GENTRAP               → calculated

// INTERRUPT               → INTERRUPT             → 0x0680

// MACHINE_CHECK           → MCHK                  → 0x0500

// RESET                   → RESET                 → 0x0780

//

// Implemented in: ExceptionMapping_inl.h::mapClassToPalVector()

 

// ============================================================================

// 6. IPR UPDATES PER EXCEPTION TYPE

// ============================================================================

//

// All exceptions:

//   - HWPCB.savedPC = faultPC

//   - HWPCB.savedPS = faultPS

//   - IPRs.exc_addr = palEntryPC

//   - HWPCB.exc_addr = palEntryPC

//

// Memory faults (ITB/DTB):

//   - IPRs.va = faultVA

//   - HWPCB.va_fault = faultVA

//   - IPRs.mm_stat = access_type | fault_type | tb_domain

//   - IPRs.exc_sum |= (1 << 0) [TB miss bit]

//

// Access violations:

//   - IPRs.exc_sum |= (1 << 1) [ACV bit]

//

// Unaligned:

//   - IPRs.exc_sum |= (1 << 2) [Unaligned bit]

//

// Arithmetic:

//   - IPRs.exc_sum |= ev.extraInfo [FP exception bits]

//

// Illegal opcode:

//   - IPRs.exc_sum |= (1 << 4) [Opcode error bit]

//

// Machine check:

//   - IPRs.mces |= error_syndrome

 

// ============================================================================

// 7. COMMON USAGE PATTERNS

// ============================================================================

 

// Pattern 1: Simple synchronous exception

void raiseDTBMiss(CPUIdType cpuId, quint64 va, bool write)

{

   auto ev = ExceptionFactory::makeDTBMissSingleEvent(cpuId, va, write);

   FaultDispatcher::queuePendingEvent(ev);

}

 

// Pattern 2: Exception with extra info (arithmetic)

void raiseArithmeticTrap(CPUIdType cpuId, quint64 pc, quint64 fpSummary)

{

   auto ev = ExceptionFactory::makeArithmeticEvent(cpuId, pc, fpSummary);

   FaultDispatcher::queuePendingEvent(ev);

}

 

// Pattern 3: CALL_PAL instruction

void executeCallPal(CPUIdType cpuId, quint64 pc, quint32 instruction)

{

   quint8 function = instruction & 0xFF;

   auto ev = ExceptionFactory::makeCallPalEvent(cpuId, pc, function);

   FaultDispatcher::queuePendingEvent(ev);

}

 

// Pattern 4: FaultDispatcher delivery

void FaultDispatcher::servicePendingEvent(CPUIdType cpuId)

{

   PendingEvent* winner = selectHighestPriority(cpuId);

   if (!winner || !winner->isValid())

       return;

   

   auto& iprs = getIPRs(cpuId);

   auto& hwpcb = getHWPCB(cpuId);

   quint64 faultPC = getFaultingPC(cpuId);

   

   preparePendingEventForDelivery(iprs, hwpcb, *winner, faultPC);

   deliverException(iprs, hwpcb, *winner);

   

   winner->clear();

}

// 10. KEY INVARIANTS

// ============================================================================

//

// MUST ALWAYS BE TRUE:

//   1. Pipeline NEVER sets palVectorId

//   2. mapClassToPalVector() is ONLY vector resolver

//   3. preparePendingEventForDelivery() is ONLY state updater

//   4. CALL_PAL vectors are CALCULATED, never stored

//   5. All ExceptionClass values handled by mapClassToPalVector()

//   6. PendingEvent.palVectorId == INVALID until preparation

//   7. ITB/DTB distinguished at ExceptionClass level

//   8. No direct EXC_ADDR/EXC_SUM/MM_STAT writes outside preparation
