// ============================================================================

// EXCB Real-World Example - Alpha Assembly with Pipeline Stages Annotated

// ============================================================================

// Use Case: Reading exception registers after a potential fault

// ============================================================================

 

/*

SCENARIO: Kernel exception handler needs to read EXC_ADDR after a DTB miss

================================================================================

 

Alpha Assembly Code:

--------------------

0x20001000:  LDQ   R1, 0(R2)      ; Load that might fault (DTB miss)

0x20001004:  ADDQ  R1, R3, R4     ; Arithmetic that might trap (overflow)

0x20001008:  STQ   R5, 8(R6)      ; Store that might fault (ACV)

0x2000100C:  EXCB                 ; Exception barrier - WAIT for faults

0x20001010:  MFPR  R7, EXC_ADDR   ; NOW safe to read exception register

0x20001014:  MFPR  R8, EXC_SUM    ; Read exception summary

0x20001018:  BEQ   R7, no_fault   ; Branch if no exception

 

KEY POINT: EXCB ensures that if ANY of the three prior instructions (LDQ, ADDQ, STQ)

caused an exception, that exception is FULLY DELIVERED before we read EXC_ADDR.

 

Without EXCB, we might read stale EXC_ADDR from a previous exception!

*/

 

// ============================================================================

// CYCLE-BY-CYCLE PIPELINE FLOW (6-Stage Pipeline)

// ============================================================================

 

/*

PIPELINE STAGES:

  0: Fetch (IF)

  1: Decode (ID)  

  2: Issue (IS)

  3: Execute (EX)

  4: Memory (MEM)

  5: Writeback (WB)

 

================================================================================

CYCLE 0: Initial State - LDQ enters pipeline

================================================================================

 

PC = 0x20001000

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | LDQ R1,0  | [empty]   | [empty]   | [empty]   | [empty]   | [empty]

       | (0x1000)  |           |           |           |           |

 

================================================================================

CYCLE 1: Pipeline filling

================================================================================

 

PC = 0x20001004

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | ADDQ R1,3 | LDQ R1,0  | [empty]   | [empty]   | [empty]   | [empty]

       | (0x1004)  | (0x1000)  |           |           |           |

 

================================================================================

CYCLE 2: Pipeline filling

================================================================================

 

PC = 0x20001008

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | STQ R5,8  | ADDQ R1,3 | LDQ R1,0  | [empty]   | [empty]   | [empty]

       | (0x1008)  | (0x1004)  | (0x1000)  |           |           |

 

================================================================================

CYCLE 3: Pipeline filling

================================================================================

 

PC = 0x2000100C

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | EXCB      | STQ R5,8  | ADDQ R1,3 | LDQ R1,0  | [empty]   | [empty]

       | (0x100C)  | (0x1008)  | (0x1004)  | (0x1000)  |           |

       

// LDQ starts executing - might cause DTB miss!

 

================================================================================

CYCLE 4: EXCB enters Decode, LDQ accesses memory

================================================================================

 

PC = 0x20001010 (BUT FETCH WILL STALL!)

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | MFPR R7   | EXCB      | STQ R5,8  | ADDQ R1,3 | LDQ R1,0  | [empty]

       | (0x1010)  | (0x100C)  | (0x1008)  | (0x1004)  | (0x1000)  |

       |           |           |           |           | ⚠️ DTB MISS!

 

// LDQ causes DTB miss in MEM stage - exception is signaled

// Fault event is queued in fault dispatcher

 

================================================================================

CYCLE 5: EXCB enters Execute - SETS STALL FLAGS

================================================================================

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | [STALL]   | MFPR R7   | EXCB      | STQ R5,8  | ADDQ R1,3 | LDQ R1,0

       |           | (0x1010)  | (0x100C)  | (0x1008)  | (0x1004)  | (0x1000)

       |           |           | ^^^^^^^^^ |           |           |

       |           |           | EXECUTES: |           |           |

       |           |           | stalled=true           |           |

       |           |           | serializeType=Barrier_EXC

       

// EXCB sets stall flags in Execute stage

// isFrontendStalled() returns TRUE - no new fetches!

// Pipeline starts draining

 

================================================================================

CYCLE 6: Pipeline draining - EXCB waits in Execute stage

================================================================================

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | [STALL]   | [empty]   | MFPR R7   | EXCB*     | STQ R5,8  | ADDQ R1,3

       |           |           | (0x1010)  | (0x100C)  | (0x1008)  | (0x1004)

       |           |           |           | (STALLED) |           |

       

// EXCB still stalled, waiting for prior instructions

// checkBarrierRelease() checks: Are prior stages clear? NO (STQ in MEM)

 

================================================================================

CYCLE 7: Still draining

================================================================================

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | [STALL]   | [empty]   | [empty]   | EXCB*     | MFPR R7   | STQ R5,8

       |           |           |           | (0x100C)  | (0x1010)  | (0x1008)

       |           |           |           | (STALLED) |           |

 

// checkBarrierRelease() checks: Are prior stages clear? NO (MFPR in MEM)

 

================================================================================

CYCLE 8: Still draining

================================================================================

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | [STALL]   | [empty]   | [empty]   | EXCB*     | [empty]   | MFPR R7

       |           |           |           | (0x100C)  |           | (0x1010)

       |           |           |           | (STALLED) |           |

 

// checkBarrierRelease() checks:

//   ✓ All prior stages clear (Fetch/Decode/Issue are empty)

//   ✓ All exceptions delivered (DTB miss was handled)

//   → RELEASE: stalled = false

 

================================================================================

CYCLE 9: EXCB released - advances to MEM stage

================================================================================

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | MFPR R8   | [empty]   | [empty]   | [empty]   | EXCB      | [empty]

       | (0x1014)  |           |           |           | (0x100C)  |

       

// EXCB advances to MEM (no-op)

// isFrontendStalled() = FALSE - fetch resumes!

// Next instruction (MFPR R8, EXC_SUM) is fetched

 

================================================================================

CYCLE 10: EXCB completes, pipeline resumed

================================================================================

 

Stage: | Fetch     | Decode    | Issue     | Execute   | MEM       | WB

-------|-----------|-----------|-----------|-----------|-----------|----------

Inst:  | BEQ R7    | MFPR R8   | [empty]   | [empty]   | [empty]   | EXCB

       | (0x1018)  | (0x1014)  |           |           |           | (0x100C)

 

// EXCB completes writeback

// EXC_ADDR is now safely readable (contains fault address from DTB miss)

// Pipeline fully operational

// Pipeline stall is released

 

================================================================================

RESULT: EXC_ADDR contains the virtual address that caused the DTB miss

        Exception handler can now process the fault correctly

================================================================================

*/

 

// ============================================================================

// CRITICAL INSIGHT: EXCB Waits for PRIOR Instructions, Not Next Ones

// ============================================================================

 

/*

CORRECT UNDERSTANDING:

=====================

 

LDQ R1, 0(R2)      ; Instruction that might fault

ADDQ R1, R3, R4    ; Instruction that might trap

STQ R5, 8(R6)      ; Instruction that might fault

EXCB               ; ← WAIT HERE until all 3 prior instructions complete

MFPR R7, EXC_ADDR  ; ← NOW safe - exception state is stable

 

The barrier ensures:

1. LDQ, ADDQ, STQ have all completed (or faulted)

2. Any exceptions they caused have been fully delivered

3. Exception registers (EXC_ADDR, EXC_SUM) reflect current state

4. MFPR reads will see correct exception information

 

*/

 

// ============================================================================

// Memory Ordering Context

// ============================================================================

 

/*

You mentioned "memory order" - that's correct but applies to memory barriers:

 

WMB (Write Memory Barrier):

---------------------------

STQ R1, 0(R2)      ; Store 1

STQ R3, 8(R4)      ; Store 2  

WMB                ; ← Ensure both stores complete in order

STQ R5, 16(R6)     ; Store 3 - won't execute until WMB releases

 

MB (Memory Barrier):

-------------------

LDQ R1, 0(R2)      ; Load 1

STQ R3, 8(R4)      ; Store 1

MB                 ; ← Ensure all prior loads/stores complete

LDQ R5, 16(R6)     ; Load 2 - won't execute until MB releases

 

EXCB (Exception Barrier):

------------------------

LDQ R1, 0(R2)      ; Might fault

ADDQ R1, R3, R4    ; Might trap

EXCB               ; ← Ensure all prior exceptions delivered

MFPR R7, EXC_ADDR  ; Safe to read exception state

*/
