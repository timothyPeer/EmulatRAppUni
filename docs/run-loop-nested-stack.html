Run Loop Nested Stack

=============================================================================

ITERATION N: Normal Mode - CALL_PAL #0x83

=============================================================================

 

AlphaCPU::runOneInstruction()

│

├─ Step 1: Check processor state

│  ├─ isPalMode = (PC[0] == 1) → FALSE

│  └─ shadowRegsActive = FALSE

│

├─ Step 2: Check CBox for pending events

│  ├─ Check interrupts (none deliverable, IPL=3)

│  └─ Check ASTs (none pending)

│

├─ Step 3: Fetch instruction

│  ├─ fetchPC = getPC_Active(cpuId) & ~0x1ULL → 0x00000001200041A0

│  ├─ Translate VA→PA via ev6TranslateFastVA()

│  ├─ Read 32 bits from memory

│  └─ rawInstruction = 0x00000083 (CALL_PAL #0x83)

│

├─ Step 4: Decode instruction

│  └─ IBox::fetchAndDecode(fetchResult)

│     ├─ grain.opcode = OPCODE_CALL_PAL (0x00)

│     ├─ grain.palFunction = 0x83

│     ├─ grain.pc = 0x00000001200041A0

│     └─ Return fetchResult.grain

│

├─ Step 5: Execute pipeline (PUMP - no side effects)

│  └─ BoxResult boxResult = AlphaPipeline::execute(fetchResult)

│     │

│     ├─ stage_IF

│     │  └─ slot.di = fetchResult.grain

│     │

│     ├─ stage_ID

│     │  └─ slot.decoded = true (no operands for CALL_PAL)

│     │

│     ├─ stage_EX (ROUTING ONLY)

│     │  ├─ Detect slot.di.opcode == OPCODE_CALL_PAL

│     │  ├─ Route to PalBox

│     │  └─ boxResult = PalBox::execute(slot)

│     │     │

│     │     ├─ PalBox::execute(slot)

│     │     │  └─ switch (slot.di.opcode)

│     │     │     └─ case OPCODE_CALL_PAL:

│     │     │        └─ return enterPal(

│     │     │              PalEntryReason::CALL_PAL_INSTRUCTION,

│     │     │              slot.di.palFunction,  // 0x83

│     │     │              slot.di.pc + 4        // 0x00000001200041A4

│     │     │           )

│     │     │

│     │     └─ PalBox::enterPal(CALL_PAL_INSTRUCTION, 0x83, 0x00000001200041A4)

│     │        │

│     │        ├─ 1. Record entry metadata

│     │        │  ├─ m_entryReason = CALL_PAL_INSTRUCTION

│     │        │  ├─ m_entryVector = 0x83

│     │        │  └─ m_faultPC = 0x00000001200041A4

│     │        │

│     │        ├─ 2. Save complete context (UNIFIED)

│     │        │  └─ saveCompleteContext(m_cpuId, m_savedContext)

│     │        │     ├─ m_savedContext.hwpcb = getHWPCB_Active(cpuId)

│     │        │     │  ├─ PC = 0x00000001200041A0

│     │        │     │  ├─ PS, IPL=3, CM=USER

│     │        │     │  ├─ KSP, ESP, SSP, USP

│     │        │     │  └─ PTBR, ASN, etc.

│     │        │     ├─ R[0..31] → m_savedContext.intRegs[0..31]

│     │        │     ├─ F[0..31] → m_savedContext.floatRegs[0..31]

│     │        │     └─ FPCR → m_savedContext.fpcr

│     │        │

│     │        ├─ 3. Compute PAL entry PC

│     │        │  ├─ palBase = getPAL_BASE_Active(cpuId) → 0xFFFFFE0000000000

│     │        │  ├─ entryPC = palBase + (0x83 * 64) → 0xFFFFFE0000002030

│     │        │  └─ (PAL function dispatch table)

│     │        │

│     │        ├─ 4. Set exception address (return address)

│     │        │  └─ setEXC_ADDR_Active(cpuId, 0x00000001200041A4) ✅ pc+4

│     │        │

│     │        ├─ 5. Enter PAL mode (CENTRALIZED - only place PC[0] set)

│     │        │  ├─ setPC_Active(cpuId, 0xFFFFFE0000002030 | 0x1)

│     │        │  │  → PC = 0xFFFFFE0000002031 (PC[0]=1, PAL mode)

│     │        │  ├─ setIPL_Active(cpuId, 7)

│     │        │  └─ setCM_Active(cpuId, CM_KERNEL)

│     │        │

│     │        ├─ 6. Activate shadow registers

│     │        │  └─ m_shadowRegsActive = true

│     │        │

│     │        └─ 7. Return BoxResult

│     │           └─ return BoxResult().requestFlush()

│     │

│     ├─ stage_MEM (no memory operation)

│     │

│     └─ stage_WB

│        └─ return boxResult (propagate BoxResult from PalBox)

│

├─ Step 6: Handle BoxResult (AlphaCPU orchestrates side effects)

│  ├─ if (boxResult.needsFlush())

│  │  └─ AlphaPipeline::flush()

│  │     ├─ Clear all pipeline stages

│  │     ├─ Invalidate prefetch buffers

│  │     └─ Clear any pending MBox/EBox state

│  │

│  └─ PC now = 0xFFFFFE0000002031 (ready for PAL fetch)

│

└─ Return to executeLoop

ITERATION N+1: PAL Mode - First PAL Instruction

=============================================================================

 

AlphaCPU::runOneInstruction()

│

├─ Step 1: Check processor state

│  ├─ isPalMode = (getPC_Active(cpuId) & 0x1) → TRUE ✅

│  └─ shadowRegsActive = TRUE

│

├─ Step 2: Check CBox (events blocked in PAL)

│  ├─ Interrupts → BLOCKED (IPL=7, highest level)

│  └─ ASTs → BLOCKED (in PAL mode)

│

├─ Step 3: Fetch instruction

│  ├─ fetchPC = getPC_Active(cpuId) & ~0x1ULL → 0xFFFFFE0000002030

│  │  (mask off PAL bit for fetch)

│  ├─ Translate VA→PA (PAL addresses identity-mapped or supervisor space)

│  └─ rawInstruction = 0x23DEFFC8 (LDA R30, -56(R30))

│

├─ Step 4: Execute pipeline

│  └─ BoxResult boxResult = AlphaPipeline::execute(fetchResult)

│     │

│     ├─ stage_EX

│     │  ├─ Route LDA to EBox

│     │  └─ EBox::executeLDA(slot)

│     │     ├─ Read R30 (normal, not shadowed)

│     │     ├─ result = R30 + (-56)

│     │     └─ slot.result = result

│     │

│     └─ stage_WB

│        ├─ Write slot.result to R30

│        └─ return BoxResult() (no flags)

│

├─ Step 5: Handle BoxResult

│  ├─ No flags set

│  ├─ Normal retirement

│  └─ PC += 4 → 0xFFFFFE0000002035 (still PC[0]=1)

│

└─ Return to executeLoop

├─ Steps 1-4: Normal fetch/decode (LDQ R1, 0(R2))

├─ Step 5: Execute pipeline

│     │     │

│     │     │  │

│     │     │  │

│     │     │  ├─ 3. Translation failed - return immediately

│     │     │  │  │  (NO memory access, NO state change!)

│     │     │  │  │

│     │     │  │        .setFaultVA(va)

│     │     │  │        .requestFlush()

│     │     │  │

│     │     │  └─ (Translation succeeded path not taken)

│     │     │

│     │     └─ BoxResult returned with fault info

│     │

│     ├─ stage_MEM (skipped - fault already detected)

│     │

│     └─ stage_WB (propagate BoxResult, no writeback on fault)

│        └─ return boxResult

│

│  │

│  ├─ if (boxResult.hasFault())

│  │  │

│  │  └─ Handle fault through unified entry

│  │     │

│  │     ├─ Extract fault info

│  │     │  ├─ faultType = DTBM_MISS

│  │     │  ├─ faultPC = 0x00000001200041C0 (LDQ instruction PC)

│  │     │  └─ faultVA = 0x0000000120FFA000

│  │     │

│  │     ├─ Determine PAL vector

│  │     │  └─ vector = DTBM_VECTOR (0xFFFFFE0000000080 or similar)

│  │     │

│  │     └─ PalBox::enterPal(DTBM_FAULT, vector, faultPC)

│  │        │

│  │        │

│  │        └─ 7. Return BoxResult

│  │

│  └─ AlphaPipeline::flush()

│     └─ Clear all stages, no stale fault state

│

FLOW 3: HW_REI - Unified Exit from PAL

=============================================================================

 

AlphaCPU::runOneInstruction()

│

├─ Step 1: Check processor state

│  ├─ isPalMode = TRUE

│  └─ shadowRegsActive = TRUE

│

├─ Step 2: Fetch HW_REI instruction

│  └─ rawInstruction = 0x7BFF8001 (HW_REI)

│

├─ Step 3: Execute pipeline

│  └─ BoxResult boxResult = AlphaPipeline::execute(fetchResult)

│     │

│     ├─ stage_EX (ROUTING ONLY)

│     │  ├─ Detect slot.di.opcode == OPCODE_HW_REI

│     │  ├─ Route to PalBox

│     │  └─ boxResult = PalBox::execute(slot)

│     │     │

│     │     ├─ PalBox::execute(slot)

│     │     │  └─ switch (slot.di.opcode)

│     │     │     └─ case OPCODE_HW_REI:

│     │     │        └─ return executeREI(slot)

│     │     │

│     │     └─ PalBox::executeREI(slot)

│     │        │

│     │        │  ├─ currentPC = getPC_Active(cpuId)

│     │        │  ├─ if (!(currentPC & 0x1))

│     │        │  │  └─ // Illegal REI - not in PAL mode!

│     │        │  │     return BoxResult()

│     │        │  │        .setFaultType(FaultType::ILLEGAL_INSTRUCTION)

│     │        │  │        .requestFlush()

│     │        │

│     │        │     │  ├─ setPS_Active(cpuId, m_savedContext.hwpcb.ps)

│     │        │     ├─ F[0..31] ← m_savedContext.floatRegs[0..31]

│     │        │

│     │        │  ├─ currentPC = getPC_Active(cpuId)

│     │        │  └─ if (currentPC & 0x1)

│     │        │     └─ setPC_Active(cpuId, currentPC & ~0x1ULL)

│     │        │

│     │

│     └─ stage_WB

│        └─ return boxResult

│

├─ Step 4: Handle BoxResult

│  ├─ AlphaPipeline::flush()

│  │  └─ Clear all stages, ensure clean state

│  │

│  └─ Determine next PC based on entry reason:

│     ├─ If entry was CALL_PAL:

│     │  └─ PC = 0x00000001200041A4 (return address, pc+4)

UNIFIED PALBOX API (CRITICAL CONTRACT)

=============================================================================

 

class PalBox {

public:

    // ========================================================================

    // SINGLE ENTRY POINT (used by both instruction and event paths)

    // ========================================================================

    BoxResult enterPal(

        PalEntryReason reason,

        quint64 vectorOrSelector,

        quint64 faultPC

    ) noexcept {

        

        // 1. Record metadata

        m_entryReason = reason;

        m_entryVector = vectorOrSelector;

        m_faultPC = faultPC;

        

        // 2. Save context (UNIFIED - same for all entry types)

        saveCompleteContext(m_cpuId, m_savedContext);

        

        // 3. Compute entry PC

        quint64 entryPC;

        if (reason == PalEntryReason::CALL_PAL_INSTRUCTION) {

            entryPC = computePalEntryPC(m_cpuId, vectorOrSelector);

        } else {

            entryPC = vectorOrSelector;  // Direct vector for faults

        }

        

        // 4. Set exception address

        setEXC_ADDR_Active(m_cpuId, faultPC);

        // For CALL_PAL: faultPC is pc+4 (return address)

        // For faults: faultPC is faulting instruction (retry address)

        

        // 5. Enter PAL mode (CENTRALIZED - only place PC[0] set)

        setPC_Active(m_cpuId, entryPC | 0x1ULL);

        setIPL_Active(m_cpuId, 7);

        setCM_Active(m_cpuId, CM_KERNEL);

        

        // 6. Activate shadow registers

        m_shadowRegsActive = true;

        

        // 7. Return flush request

        return BoxResult().requestFlush();

    }

    

    // ========================================================================

    // SINGLE EXIT POINT (unified for all return types)

    // ========================================================================

    BoxResult executeREI(const PipelineSlot& slot) noexcept {

        

        // 1. Validate in PAL mode

        if (!(getPC_Active(m_cpuId) & 0x1)) {

            // Illegal REI outside PAL mode

            return BoxResult()

                .setFaultType(FaultType::ILLEGAL_INSTRUCTION)

                .requestFlush();

        }

        

        // 2. Restore context (UNIFIED - restores everything)

        restoreCompleteContext(m_cpuId, m_savedContext);

        

        // 3. Deactivate shadow registers

        m_shadowRegsActive = false;

        

        // 4. Ensure PC[0] clear (double-check)

        quint64 pc = getPC_Active(m_cpuId);

        if (pc & 0x1) {

            setPC_Active(m_cpuId, pc & ~0x1ULL);

        }

        

        // 5. Return flush request

        return BoxResult().requestFlush();

    }

    

    // ========================================================================

    // GRAIN EXECUTOR (routes to entry/exit based on opcode)

    // ========================================================================

    BoxResult execute(const PipelineSlot& slot) noexcept {

        switch (slot.di.opcode) {

            case OPCODE_CALL_PAL:

                return enterPal(

                    PalEntryReason::CALL_PAL_INSTRUCTION,

                    slot.di.palFunction,

                    slot.di.pc + 4  // Return to next instruction

                );

            

            case OPCODE_HW_REI:

                return executeREI(slot);

            

            case OPCODE_HW_MFPR:

                return executeHW_MFPR(slot);

            

            case OPCODE_HW_MTPR:

                return executeHW_MTPR(slot);

            

            default:

                return BoxResult();  // Unknown opcode

        }

    }

 

private:

    CPUIdType m_cpuId;

    CPUContextSnapshot m_savedContext;  // Complete context storage

    PalEntryReason m_entryReason;

    quint64 m_entryVector;

    quint64 m_faultPC;

    bool m_shadowRegsActive;

};

ALPHACPU ORCHESTRATION (FINAL PATTERN)

=============================================================================

 

void AlphaCPU::runOneInstruction() {

    

    // 1. Check for external events (interrupts, IPIs)

    if (checkForInterrupt()) {

        m_palbox->enterPal(

            PalEntryReason::INTERRUPT,

            INTERRUPT_VECTOR,

            getPC_Active(m_cpuId)

        );

        m_pipeline->flush();

        return;

    }

    

    // 2. Fetch and decode

    FetchResult fetchResult = fetchInstruction();

    

    // 3. Execute pipeline (PUMP)

    BoxResult boxResult = m_pipeline->execute(fetchResult);

    

    // 4. Handle BoxResult flags

    

    // 4a. Fault detected?

    if (boxResult.hasFault()) {

        quint64 vector = getFaultVector(boxResult.faultType());

        

        m_palbox->enterPal(

            getFaultReason(boxResult.faultType()),

            vector,

            boxResult.faultPC()  // Faulting instruction PC

        );

        

        m_pipeline->flush();

        return;

    }

    

    // 4b. Flush requested?

    if (boxResult.needsFlush()) {

        m_pipeline->flush();

    }

    

    // 4c. Other side effects

    if (boxResult.needsMemoryBarrier()) {

        // Handle memory barrier

    }

    

    // 5. Normal retirement

    // PC already updated by pipeline or PalBox

}

   - Faults save complete context

   - REI restores complete context

   - No drift possible

   - Translation BEFORE memory access

   - Translation BEFORE register write

   - Fault returns immediately, no state change

   - All stages cleared

   - No stale fault bits

   - No pending MBox operations

 

✅ 6. REI validates PAL mode

   - Generate fault if not in PAL mode

   - Prevents illegal state transitions
